	//FUNCTION PROTOTYPE
	
		/*	
		 *Funtion transforms a number from binary to decimal		
		 */
			int transf_2v10(int binarna){

				int result = 0,i = 0,bit;
				while (binarna != 0){
					bit = binarna%10;
					binarna = binarna/10;
					result += bit*pow(2,i);
					i++;
				}
				return result;
			}
