		/*
		 *FROM NUMBER TO ASCII
		 *Function reads a number int a 
		 *Transforms it to ASCII code 
		 *returns an pointer to global variable char result
		 *
		 */
				char* to_array (int a){
					int j = 0;
					while (j<4){
						result[j] = 0;
						j++;
					}
					int i = (int)(log10(a));
					while (a != 0){
						result[i] = a%10 + 48;
						a = a/10;
						i--;
					}
					return result;
				}
